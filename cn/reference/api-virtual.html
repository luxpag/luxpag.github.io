# 虚拟账号接口

# 请求接口ROOT地址

  测试环境 : https://dev-gateway.luxpag.com
  正式环境 : https://gateway.luxpag.com

# 接口

  虚拟账号创建：/trade/account-create
  虚拟账号取消：/trade/account-cancel
  虚拟账号查询：/trade/account-query

# Request Header

参数名 必须 描述
Content-Type 推荐 application/json
Authorization yes Basic Base64(app_id:secret_key)

# 请求参数（JSON格式）

# 创建虚拟账号

参数名 类型 必须 最大长度/默认值 描述
app_id string yes 32 商户后台创建并获得
timestamp string yes 19 yyyy-MM-dd HH:mm:ss
notify_url string yes 回调地址,IPN通知地址
customer.buyer_id string yes 商户的用户ID
customer.email string yes 用户邮箱（不支持‘+’号）
customer.name string no 用户姓名
customer.phone string no 用户电话
customer.identify.number string no 用户ID
customer.identify.type string no 用户ID类型

# 查询虚拟账号

参数名 类型 必须 最大长度/默认值 描述
app_id string yes 32 商户后台创建并获得
timestamp string yes 19 yyyy-MM-dd HH:mm:ss
account_id string yes 虚拟账号ID

# 取消虚拟账号

参数名 类型 必须 最大长度/默认值 描述
app_id string yes 32 商户后台创建并获得
timestamp string yes 19 yyyy-MM-dd HH:mm:ss
account_id string yes 虚拟账号ID

# 请求样例

curl --location --request POST 'https://gateway.luxpag.com/trade/account-create' \
--header 'Authorization: Basic Base64(appid:secret_key)' \
--header 'Content-Type: application/json' \
--data-raw '{
    "app_id": "app_id",
    "notify_url": "notify_url",
    "timestamp": "{{datetime}}",
    "customer": {
      "email": "email",
      "buyer_id": "buyer_id"
    }
}'

curl --location --request POST 'https://gateway.luxpag.com/trade/account-query' \
--header 'Authorization: Basic Base64(appid:secret_key)' \
--header 'Content-Type: application/json' \
--data-raw '{
    "app_id": "app_id",
    "account_id": "account_id",
    "timestamp": "{{datetime}}"
}'

curl --location --request POST 'https://gateway.luxpag.com/trade/account-cancel' \
--header 'Authorization: Basic Base64(appid:secret_key)' \
--header 'Content-Type: application/json' \
--data-raw '{
    "app_id": "app_id",
    "account_id": "account_id",
    "timestamp": "{{datetime}}"
}'

# 响应（JSON格式）

参数名 类型 描述
code string 返回码
msg string 返回消息
sub_code string 子码
sub_msg string 子消息
account_id string 虚拟账号ID
account_number string 虚拟账号
provider string 渠道
status string 状态；ACTIVE、CANCELED
notify_url string 回调地址,IPN通知地址
beneficiary_name string 服务商

# 返回样例（成功）

{
    "code": "10000",
    "msg": "Success",
    "account_id": "{account_id}",
    "account_number": "{account_number}",
    "account_number": "{account_number}",
    "provider": "{provider}",
    "status": "{status}",
    "notify_url": "{notify_url}",
}

# 返回样例（失败）

{
    "code": "40002",
    "msg": "Business Failed",
    "sub_code": "invalid-signature",
    "sub_msg": "invalid signature"
}
